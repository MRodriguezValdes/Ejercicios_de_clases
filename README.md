import math


def merge_sort(arr: list[int]):
    if len(arr) > 1:
        middle: int = len(arr) // 2
        L: list[int] = arr[:middle]
        R: list[int] = arr[middle:]

        merge_sort(L)
        merge_sort(R)

        i, j, k = 0, 0, 0

        while i < len(L) and j < len(R):
            if L[i] <= R[j]:
                arr[k] = L[i]
                i += 1
            else:
                arr[k] = R[j]
                j += 1
            k += 1
        while i < len(L):
            arr[k] = L[i]
            i += 1
            k += 1
        while j < len(R):
            arr[k] = R[j]
            j += 1
            k += 1


def is_primo(number: int) -> bool:
    for i in range(2, int(math.sqrt(number)) + 1):
        if number % i == 0:
            return False

    return True
